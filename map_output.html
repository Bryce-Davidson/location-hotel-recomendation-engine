def find_bounding_box_center(df: pd.DataFrame):
    lat_min, lat_max = df['lat'].min(), df['lat'].max()
    lng_min, lng_max = df['lng'].min(), df['lng'].max()
    center_lat = (lat_min + lat_max) / 2
    center_lng = (lng_min + lng_max) / 2
    return [center_lat, center_lng]

# create a map
combined = pd.concat([df_locations, df_hotels]).reset_index(drop=True)
center = find_bounding_box_center(combined)

# disable controls
m = folium.Map(location=center, zoom_start=12, zoom_control=False)

# add markers to the map
for i, row in df_locations.iterrows():
    folium.CircleMarker(
        location=[row['lat'], row['lng']],
        radius=3,
        popup=row['name'],
        stroke=True,
        fill=True,
        fill_color='blue',
        color='white',
        weight=1,
        fill_opacity=0.7
